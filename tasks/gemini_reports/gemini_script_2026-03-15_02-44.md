        if self._client is None:
            from groq import Groq
            from config.config_loader import config

            api_key = getattr(config, "groq_api_key", None) or getattr(
                config, "groq_key", None
            )
            self._client = Groq(api_key=api_key)
        return self._client

    def _get_groq_model(self) -> str:
        try:
            from config.config_loader import config

            return config.groq_config.get("model", "llama-3.3-70b-versatile")
        except Exception:
            return "llama-3.3-70b-versatile"

    # ─── Catálogo de modos ─────────────────────────────────────────────────────

    def get_modes(self) -> dict:
        """Devuelve el catálogo completo: 14 modos de content_modes + 2 de v4."""
        from backend.services.content_modes import ALL_MODES

        return {**ALL_MODES, **_V4_ONLY_MODES}

    def _get_mode_config(self, mode: str) -> dict:
        """Devuelve la config del modo (system, temperature, emoji, etc.)."""
        from backend.services.content_modes import ALL_MODES

        if mode in ALL_MODES:
            return ALL_MODES[mode]
        if mode in _V4_ONLY_MODES:
            return _V4_ONLY_MODES[mode]
        # fallback a historian
        return ALL_MODES["historian"]

    # ─── System prompt ─────────────────────────────────────────────────────────

    def _get_system_prompt(self, mode: str, language: str) -> str:
        """
        Construye el system prompt para el modo e idioma dados.
        - Modos de content_modes → usa su 'system' completo (con ejemplos embebidos)
        - Modos v4-only → usa su system prompt de v4
        En ambos casos inyecta la instrucción de idioma al final.

        base = mode_cfg["system"]

            base + f"\n\nCRITICAL: You MUST write the entire script in {lang_name}. "
            f"Every single word must be in {lang_name}."
        )

                f"Write a {duracion}-second video script in your style.\n"

                f"EXAMPLE:\n{example}\n\n"
                f"Now write a script of similar length about: {tema}\n"
                f"Write ONLY the script. No labels, no quotes, no stage directions."
            )

                f"You MUST write at least {target_words} words. "
